```typescript
function composeFloatPlanTemplate(params: {
  tripName: string;
  startDate: string;
  endDate: string;
  destinationName: string | null;
  vehicleName: string | null;
  notes: string | null;
  mapsLink: string | null;
}): { subject: string; body: string } {
  const destination = params.destinationName || params.tripName;
  const startFormatted = params.startDate;
  const endFormatted = params.endDate;

  const lines: string[] = [
    `Will is camping at ${destination} from ${startFormatted} to ${endFormatted}.`,
  ];

  if (params.vehicleName) {
    lines.push(`Vehicle: ${params.vehicleName}.`);
  }

  lines.push(`Expected return: ${endFormatted}.`);

  if (params.notes) {
    lines.push(`Notes: ${params.notes}`);
  }

  if (params.mapsLink) {
    lines.push(`Map: ${params.mapsLink}`);
  }

  return {
    subject: `Float Plan: ${params.tripName} (${startFormatted})`,
    body: lines.join('\n'),
  };
}
```
