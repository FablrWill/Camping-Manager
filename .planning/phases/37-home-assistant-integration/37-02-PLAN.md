   import { NextResponse } from 'next/server';
   import { prisma } from '@/lib/db';
   import { SETTINGS_ID, HA_TIMEOUT_MS } from '@/lib/ha';

   export async function POST(): Promise<NextResponse> {
     try {
       const settings = await prisma.settings.findUnique({ where: { id: SETTINGS_ID } });
       if (!settings?.haUrl || !settings?.haToken) {
         return NextResponse.json({ error: 'Home Assistant not configured' }, { status: 400 });
       }
       // Step 1: Health check GET /api/
       const healthRes = await fetch(`${settings.haUrl}/api/`, {
         headers: { Authorization: `Bearer ${settings.haToken}`, 'Content-Type': 'application/json' },
         signal: AbortSignal.timeout(HA_TIMEOUT_MS),
       });
       if (!healthRes.ok) {
         return NextResponse.json({ error: `Home Assistant returned ${healthRes.status}` }, { status: 502 });
       }
       // Step 2: Get entity count from GET /api/states
       const statesRes = await fetch(`${settings.haUrl}/api/states`, {
         headers: { Authorization: `Bearer ${settings.haToken}`, 'Content-Type': 'application/json' },
         signal: AbortSignal.timeout(HA_TIMEOUT_MS),
       });
       if (!statesRes.ok) {
         return NextResponse.json({ error: `Failed to fetch entities: ${statesRes.status}` }, { status: 502 });
       }
       const states = await statesRes.json();
       return NextResponse.json({ connected: true, entityCount: Array.isArray(states) ? states.length : 0 });
     } catch (error) {
       console.error('HA test failed:', error);
       return NextResponse.json({ error: 'Could not reach Home Assistant' }, { status: 502 });
     }
   }
   ```

   import { NextResponse } from 'next/server';
   import { prisma } from '@/lib/db';
   import { SETTINGS_ID, HA_TIMEOUT_MS, HAEntityState } from '@/lib/ha';

   export async function GET(): Promise<NextResponse> {
     try {
       const settings = await prisma.settings.findUnique({ where: { id: SETTINGS_ID } });
       if (!settings?.haUrl || !settings?.haToken) {
         return NextResponse.json({ error: 'Home Assistant not configured' }, { status: 400 });
       }
       const res = await fetch(`${settings.haUrl}/api/states`, {
         headers: { Authorization: `Bearer ${settings.haToken}`, 'Content-Type': 'application/json' },
         signal: AbortSignal.timeout(HA_TIMEOUT_MS),
       });
       if (!res.ok) {
         return NextResponse.json({ error: `Home Assistant returned ${res.status}` }, { status: 502 });
       }
       const entities: HAEntityState[] = await res.json();
       // Group by domain prefix (entity_id format: "domain.object_id")
       const domains: Record<string, Array<{ entity_id: string; friendly_name: string; state: string; unit: string }>> = {};
       for (const entity of entities) {
         const domain = entity.entity_id.split('.')[0];
         const entry = {
           entity_id: entity.entity_id,
           friendly_name: entity.attributes.friendly_name ?? entity.entity_id,
           state: entity.state,
           unit: entity.attributes.unit_of_measurement ?? '',
         };
         domains[domain] = [...(domains[domain] ?? []), entry];
       }
       return NextResponse.json({ domains });
     } catch (error) {
       console.error('HA entities fetch failed:', error);
       return NextResponse.json({ error: 'Could not reach Home Assistant' }, { status: 502 });
     }
   }
   ```

   ```typescript
   import { NextResponse } from 'next/server';
   import { prisma } from '@/lib/db';
   import { SETTINGS_ID, HA_TIMEOUT_MS, HAEntityState, CachedEntityState, safeParseEntityIds } from '@/lib/ha';

   export async function GET(): Promise<NextResponse> {
     try {
       const settings = await prisma.settings.findUnique({ where: { id: SETTINGS_ID } });
       if (!settings?.haUrl || !settings?.haToken) {
         return NextResponse.json({ error: 'Home Assistant not configured' }, { status: 400 });
       }

       const entityIds = safeParseEntityIds(settings.haEntityIds ?? null);
       if (entityIds.length === 0) {
         return NextResponse.json({ entities: [], stale: false, lastFetched: null });
       }

       try {
         // Fetch all states then filter to selected entities
         const res = await fetch(`${settings.haUrl}/api/states`, {
           headers: { Authorization: `Bearer ${settings.haToken}`, 'Content-Type': 'application/json' },
           signal: AbortSignal.timeout(HA_TIMEOUT_MS),
         });

         if (!res.ok) throw new Error(`HA returned ${res.status}`);

         const allStates: HAEntityState[] = await res.json();
         const selectedSet = new Set(entityIds);
         const entities: CachedEntityState[] = allStates
           .filter(s => selectedSet.has(s.entity_id))
           .map(s => ({
             entityId: s.entity_id,
             friendlyName: s.attributes.friendly_name ?? s.entity_id,
             state: s.state,
             unit: s.attributes.unit_of_measurement ?? '',
           }));

         // Preserve selection order from entityIds
         const ordered = entityIds
           .map(id => entities.find(e => e.entityId === id))
           .filter((e): e is CachedEntityState => e !== undefined);

         // Update cache (D-10)
         const now = new Date();
         await prisma.settings.update({
           where: { id: SETTINGS_ID },
           data: {
             haEntityCache: JSON.stringify(ordered),
             haLastFetched: now,
           },
         });

         return NextResponse.json({ entities: ordered, stale: false, lastFetched: now.toISOString() });
       } catch (haError) {
         // HA unreachable — fallback to cache (D-09)
         console.error('HA states fetch failed, using cache:', haError);
         const cached = settings.haEntityCache
           ? ((): CachedEntityState[] => { try { return JSON.parse(settings.haEntityCache); } catch { return []; } })()
           : [];
         return NextResponse.json({
           entities: cached,
           stale: true,
           lastFetched: settings.haLastFetched?.toISOString() ?? null,
         });
       }
     } catch (error) {
       console.error('HA states route error:', error);
       return NextResponse.json({ error: 'Failed to fetch HA states' }, { status: 500 });
     }
   }
   ```
