**Test Connection handler:**
```typescript
const handleTest = async () => {
  setTesting(true);
  setTestResult(null);
  try {
    // Save URL + token first so the server has them for test
    await fetch('/api/settings', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ haUrl, ...(haToken ? { haToken } : {}) }),
    });
    const res = await fetch('/api/ha/test', { method: 'POST' });
    const data = await res.json();
    if (res.ok && data.connected) {
      setTestResult({ success: true, message: `Connected — ${data.entityCount} entities found` });
    } else {
      setTestResult({ success: false, message: data.error ?? 'Unknown error' });
    }
  } catch {
    setTestResult({ success: false, message: 'Could not reach Home Assistant — check URL and token' });
  } finally {
    setTesting(false);
  }
};
```
