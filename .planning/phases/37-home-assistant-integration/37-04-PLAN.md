**Polling logic (per D-03, Pattern 3 from RESEARCH.md):**
```typescript
useEffect(() => {
  let mounted = true;
  const fetchStates = async () => {
    try {
      const res = await fetch('/api/ha/states');
      if (!res.ok) throw new Error('fetch failed');
      const data = await res.json();
      if (!mounted) return;
      setEntities(data.entities);
      setStale(data.stale);
      setLastFetched(data.lastFetched);
      setLoading(false);
      setError(data.entities.length === 0 && data.stale);
    } catch {
      if (!mounted) return;
      setStale(true);
      setLoading(false);
      if (entities.length === 0) setError(true);
    }
  };
  fetchStates(); // immediate first fetch
  const interval = setInterval(fetchStates, 30_000);
  return () => { mounted = false; clearInterval(interval); };
}, []); // empty deps — Pitfall 4 prevention
```

```typescript
function getDomainIcon(entityId: string) {
  const domain = entityId.split('.')[0];
  switch (domain) {
    case 'sensor': return <Thermometer size={18} />;
    case 'binary_sensor': return <Activity size={18} />;
    case 'device_tracker': return <MapPin size={18} />;
    case 'switch': return <Power size={18} />;
    default: return <Wifi size={18} />;
  }
}
```

**Relative time helper** (inline):
```typescript
function timeAgo(iso: string): string {
  const seconds = Math.floor((Date.now() - new Date(iso).getTime()) / 1000);
  if (seconds < 60) return 'just now';
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  return `${hours}h ago`;
}
```
