Add a GET handler that reads saved packing list from Trip AND packed state from PackingItem:
```typescript
export async function GET(request: NextRequest) {
  try {
    const tripId = request.nextUrl.searchParams.get('tripId');
    if (!tripId) {
      return NextResponse.json({ error: 'tripId is required' }, { status: 400 });
    }
    const trip = await prisma.trip.findUnique({
      where: { id: tripId },
      select: {
        packingListResult: true,
        packingListGeneratedAt: true,
        packingItems: {
          select: { gearId: true, packed: true, notes: true },
        },
      },
    });
    if (!trip) {
      return NextResponse.json({ error: 'Trip not found' }, { status: 404 });
    }
    // Build a map of gearId -> packed state for the client to initialize checkboxes
    const packedState: Record<string, boolean> = {};
    for (const item of trip.packingItems) {
      packedState[item.gearId] = item.packed;
    }
    return NextResponse.json({
      result: trip.packingListResult ? JSON.parse(trip.packingListResult) : null,
      generatedAt: trip.packingListGeneratedAt?.toISOString() ?? null,
      packedState,
    });
  } catch (error) {
    console.error('Failed to fetch packing list:', error);
    return NextResponse.json({ error: 'Failed to fetch packing list' }, { status: 500 });
  }
}
```

**Addresses review concern #4 (HIGH): D-04 packed state reset race condition.**
Update the existing POST handler to use `prisma.$transaction` for atomic reset + persist. The old approach reset packed state BEFORE generation, meaning a timeout would lose state with no new list. The correct approach: generate first, then atomically reset + persist only after success.

```typescript
// In POST handler, AFTER generatePackingList() succeeds:
try {
  const packingList = await generatePackingList({ ... });

  // Atomic: reset packed state + upsert PackingItems + save result to Trip
  // Only runs if generation succeeded -- no race condition
  await prisma.$transaction(async (tx) => {
    // D-04: Reset all packed state for this trip (new list = clean slate)
    await tx.packingItem.updateMany({
      where: { tripId },
      data: { packed: false },
    });

    // Upsert PackingItems from the generated list
    for (const category of packingList.categories) {
      for (const item of category.items) {
        if (item.gearId) {
          await tx.packingItem.upsert({
            where: { tripId_gearId: { tripId, gearId: item.gearId } },
            create: { tripId, gearId: item.gearId, packed: false },
            update: { packed: false },  // Reset per D-04
          });
        }
      }
    }

    // Persist the full JSON result to Trip
    await tx.trip.update({
      where: { id: tripId },
      data: {
        packingListResult: JSON.stringify(packingList),
        packingListGeneratedAt: new Date(),
      },
    });
  });

  return NextResponse.json(packingList);
} catch (error) {
  const message = error instanceof Error ? error.message : 'Failed to generate packing list';
  if (message.includes('schema mismatch') || message.includes('non-JSON')) {
    return NextResponse.json({ error: message }, { status: 422 });
  }
  console.error('Failed to generate packing list:', error);
  return NextResponse.json({ error: 'Failed to generate packing list' }, { status: 500 });
}
```

export async function PUT(request: NextRequest) {
  try {
