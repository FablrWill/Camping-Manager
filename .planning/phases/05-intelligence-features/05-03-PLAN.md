    ```typescript
    function getSupportedMimeType(): string {
      const types = ['audio/webm;codecs=opus', 'audio/webm', 'audio/mp4', 'audio/ogg;codecs=opus']
      for (const type of types) {
        if (MediaRecorder.isTypeSupported(type)) return type
      }
      return ''
    }
    ```
