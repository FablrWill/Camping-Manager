model InboxItem {
  id          String   @id @default(cuid())
  sourceType  String   // "image" | "url" | "text"
  rawContent  String   // original URL or text; empty string for image-only
  status      String   @default("pending") // "pending" | "accepted" | "rejected"
  triageType  String?  // "gear" | "location" | "knowledge" | "tip" | "unknown"
  suggestion  String?  // JSON: typed per triageType (GearSuggestion | LocationSuggestion | KnowledgeSuggestion)
  summary     String?  // one-line AI description for inbox card display
  confidence  Float?   // 0.0-1.0 AI confidence score
  imagePath   String?  // /photos/inbox/[cuid].jpg (compressed copy)
  createdAt   DateTime @default(now())
  processedAt DateTime?

  @@index([status])
  @@index([createdAt])
}
```

// --- Inbox Triage Schemas (Phase 24) ---

export const GearSuggestionSchema = z.object({
  triageType: z.literal('gear'),
  name: z.string(),
  brand: z.string().optional(),
  category: z.string(),
  description: z.string().optional(),
  price: z.number().optional(),
  purchaseUrl: z.string().optional(),
  isWishlist: z.boolean().default(true),
})

export type GearSuggestion = z.infer<typeof GearSuggestionSchema>

export const LocationSuggestionSchema = z.object({
  triageType: z.literal('location'),
  name: z.string(),
  latitude: z.number().optional(),
  longitude: z.number().optional(),
  description: z.string().optional(),
  type: z.string().optional(),
})

export type LocationSuggestion = z.infer<typeof LocationSuggestionSchema>

export const KnowledgeSuggestionSchema = z.object({
  triageType: z.literal('knowledge'),
  content: z.string(),
  title: z.string(),
})

export type KnowledgeSuggestion = z.infer<typeof KnowledgeSuggestionSchema>

export const TipSuggestionSchema = z.object({
  triageType: z.literal('tip'),
  content: z.string(),
  title: z.string(),
})

export type TipSuggestion = z.infer<typeof TipSuggestionSchema>

export const SuggestionSchema = z.discriminatedUnion('triageType', [
  GearSuggestionSchema,
  LocationSuggestionSchema,
  KnowledgeSuggestionSchema,
  TipSuggestionSchema,
])

export type Suggestion = z.infer<typeof SuggestionSchema>

export const GearFromImageResponseSchema = z.object({
  name: z.string(),
  brand: z.string().optional(),
  category: z.string(),
  description: z.string().optional(),
  confidence: z.number().min(0).max(1),
})

export const GearFromUrlResponseSchema = z.object({
  name: z.string(),
  brand: z.string().optional(),
  category: z.string(),
  description: z.string().optional(),
  price: z.number().optional(),
  confidence: z.number().min(0).max(1),
})

export const LocationFromImageResponseSchema = z.object({
  name: z.string(),
  description: z.string().optional(),
  type: z.string().optional(),
  confidence: z.number().min(0).max(1),
})

export const ClassifyTextResponseSchema = z.object({
  triageType: z.enum(['gear', 'location', 'knowledge', 'tip', 'unknown']),
  confidence: z.number().min(0).max(1),
  summary: z.string(),
  title: z.string().optional(),
  content: z.string().optional(),
})
```

1. Create lib/intake/triage.ts:
   - Export `type TriageType = 'gear' | 'location' | 'knowledge' | 'tip' | 'unknown'`
   - Export `interface TriageResult { triageType: TriageType; confidence: number; summary: string; suggestion: GearSuggestion | LocationSuggestion | KnowledgeSuggestion | TipSuggestion | null }`
   - Export `interface TriageInput { text?: string; url?: string; imageBuffer?: Buffer; imageMime?: string }`
   - Export `async function triage(input: TriageInput): Promise<TriageResult>`
   - Routing logic:
     - If imageBuffer exists: call extractGps(imageBuffer) from lib/exif.ts. If GPS data found, call locationFromImage(imageBuffer, imageMime, exifData). Otherwise call gearFromImage(imageBuffer, imageMime).
     - If url exists: call gearFromUrl(url)
     - If text exists: call classifyText(text)
     - Otherwise: return `{ triageType: 'unknown', confidence: 0, summary: 'No input provided', suggestion: null }`

2. Create lib/intake/extractors/gear-from-url.ts:
   - Export `async function gearFromUrl(url: string): Promise<TriageResult>`
   - Fetch URL with `User-Agent: 'OutlandOS/1.0'` header, timeout 10 seconds via AbortController
   - Load HTML with `cheerio.load(html)`
   - Extract raw text: title, h1, meta description, product description divs — concatenate into rawText (max 2000 chars)
   - If rawText < 100 chars, return low confidence result: `{ triageType: 'gear', confidence: 0.2, summary: 'URL saved - product details need manual review', suggestion: { triageType: 'gear', name: url, category: 'tools', isWishlist: true } }`
   - Otherwise call Claude Haiku (`claude-haiku-4-20250514`) with prompt asking to extract gear fields from the scraped text. Include the 15 category values from lib/gear-categories.ts in the prompt so Claude picks the right one.
   - Parse response with `parseClaudeJSON(response, GearFromUrlResponseSchema)`
   - Map to TriageResult with GearSuggestion (set isWishlist: true, purchaseUrl: url)

3. Create lib/intake/extractors/gear-from-image.ts:
   - Export `async function gearFromImage(imageBuffer: Buffer, imageMime: string): Promise<TriageResult>`
   - Convert imageBuffer to base64: `imageBuffer.toString('base64')`
   - Call Claude Sonnet (`claude-sonnet-4-20250514`) with vision message: image as base64 source + text prompt asking to identify camping/outdoor gear in the image. Include the 15 category values in the prompt.
   - Parse response with `parseClaudeJSON(response, GearFromImageResponseSchema)`
   - Map to TriageResult with GearSuggestion (set isWishlist: true)

4. Create lib/intake/extractors/location-from-image.ts:
   - Export `async function locationFromImage(imageBuffer: Buffer, imageMime: string, exif: ExifGpsData): Promise<TriageResult>`
   - Call Claude Sonnet with vision message asking to describe the camping/outdoor location shown
   - Parse response with `parseClaudeJSON(response, LocationFromImageResponseSchema)`
   - Map to TriageResult with LocationSuggestion — include exif.latitude, exif.longitude from the EXIF data (not from Claude's response)

5. Create lib/intake/extractors/classify-text.ts:
   - Export `async function classifyText(text: string): Promise<TriageResult>`
   - Call Claude Haiku (`claude-haiku-4-20250514`) with prompt: classify this text as gear recommendation, location info, camping knowledge, or practical tip
   - Parse response with `parseClaudeJSON(response, ClassifyTextResponseSchema)`
   - Map to appropriate suggestion type based on triageType field
