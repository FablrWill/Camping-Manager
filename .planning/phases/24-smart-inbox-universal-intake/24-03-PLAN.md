   import InboxClient from '@/components/InboxClient'

   export default async function InboxPage() {
     const items = await prisma.inboxItem.findMany({

     })
     return <InboxClient initialItems={items} />

2. Create components/InboxClient.tsx as 'use client' component:

   State:
   - `items`: InboxItem[] (initialized from props)
   - `filter`: 'pending' | 'accepted' | 'rejected' | 'all' (default 'pending')
   - `acceptingItem`: InboxItem | null (item being accepted — opens form modal)
   - `error`: string | null

   Define InboxItem interface matching the Prisma model fields (id, sourceType, rawContent, status, triageType, suggestion, summary, confidence, imagePath, createdAt, processedAt).

   Render:
   - Filter tabs at top: "Pending", "Accepted", "Rejected", "All" — each tab fetches via `/api/inbox?status=X` (or all for "All")
   - Card list: for each item, render a card with:
     - Source type icon: Camera icon for 'image', Link icon for 'url', FileText icon for 'text' (use lucide-react icons)
     - Triage type badge: colored pill showing triageType (amber for gear, green for location, blue for knowledge, purple for tip, gray for unknown)
     - Summary text (item.summary)
     - Confidence indicator: show as percentage text with color coding (green >= 0.7, amber >= 0.4, red < 0.4)
     - Timestamp: relative time from item.createdAt
     - Image thumbnail if imagePath exists: `<img src={item.imagePath} />`
     - Raw content preview: truncated rawContent (first 100 chars)
   - Action buttons per card (only for status === 'pending'):
     - "Accept" button (green): onClick calls handleAccept(item)
     - "Reject" button (red outline): onClick calls handleReject(item.id)

   handleAccept(item):
   - Parse item.suggestion from JSON
   - If triageType === 'gear':
     - Set acceptingItem = item (opens GearForm modal)
   - If triageType === 'location':
     - For now, call accept API directly (no LocationForm pre-fill — locations are simpler)
     - `fetch('/api/inbox/${item.id}/accept', { method: 'POST' })`
     - Remove item from list on success
   - If triageType === 'knowledge' or 'tip':
     - Call accept API directly (server-side RAG ingest, no UI form needed)
     - Remove item from list on success

   GearForm modal (when acceptingItem is set):
   - Parse suggestion JSON to get gear fields
   - Build a partial GearItem from suggestion: `{ id: '', name: suggestion.name, brand: suggestion.brand ?? null, category: suggestion.category, description: suggestion.description ?? null, ... }` with null for fields not in suggestion (condition, weight, photoUrl, storageLocation, notes, wattage, hoursPerDay, hasBattery, modelNumber, connectivity, manualUrl)
   - Set isWishlist from suggestion
   - Render in a Modal with GearForm:
     ```tsx
     <GearForm
       item={partialGearItem}
       conditions={CONDITIONS}
       onSave={async (data) => {
         // First create gear via existing API
         const res = await fetch('/api/gear', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
         if (!res.ok) throw new Error('Failed to create gear')
         // Then mark inbox item accepted
         await fetch(`/api/inbox/${acceptingItem.id}/accept`, { method: 'POST' })
         // Remove from list and close modal
         setItems(prev => prev.filter(i => i.id !== acceptingItem.id))
         setAcceptingItem(null)
       }}
       onClose={() => setAcceptingItem(null)}
     />
     ```
   - Define CONDITIONS array locally matching GearClient.tsx pattern: `[{ value: 'new', label: 'New' }, { value: 'good', label: 'Good' }, { value: 'fair', label: 'Fair' }, { value: 'worn', label: 'Worn' }, { value: 'broken', label: 'Broken' }] as const`

   handleReject(id: string):
   - `fetch('/api/inbox/${id}/reject', { method: 'POST' })`
   - On success: `setItems(prev => prev.filter(i => i.id !== id))`

   Empty state: "No items in inbox" message when filtered list is empty.

   Style with Tailwind following existing dark mode patterns: stone-800/stone-900 backgrounds, amber accents.
