**lib/agent/memory.ts** — Context windowing (D-14) + memory extraction (D-13):

```typescript
import { prisma } from '@/lib/db'

export const SLIDING_WINDOW_SIZE = 20

export async function buildContextWindow(
  conversationId: string | null,
  pageContext?: string
) {
  // Type: returns array of { role: 'user' | 'assistant', content: string }
  const messages: Array<{ role: 'user' | 'assistant'; content: string }> = []

  // Load agent memories
  const memories = await prisma.agentMemory.findMany()
  const memoryBlock = memories.length > 0
    ? `\n\nWHAT I KNOW ABOUT YOU:\n${memories.map(m => `- ${m.key}: ${m.value}`).join('\n')}`
    : ''

  const contextBlock = pageContext ? `\nCURRENT CONTEXT: ${pageContext}` : ''

  // Inject memory + context as synthetic exchange to maintain alternating roles
  if (memoryBlock || contextBlock) {
    messages.push(
      { role: 'user', content: `[System context]${memoryBlock}${contextBlock}` },
      { role: 'assistant', content: 'Got it.' }
    )
  }

  // Load recent conversation history if conversationId exists
  if (conversationId) {
    const recent = await prisma.message.findMany({
      where: { conversationId },
      orderBy: { createdAt: 'desc' },
      take: SLIDING_WINDOW_SIZE,
    })

    const historyMessages = recent
      .reverse()
      .map(m => ({ role: m.role as 'user' | 'assistant', content: m.content }))

    messages.push(...historyMessages)
  }

  return messages
}

export async function extractAndSaveMemory(assistantMessage: string, userMessage: string): Promise<void> {
  // Simple heuristic: look for memory-worthy patterns in the conversation
  // Memory extraction runs fire-and-forget after each response
  const memoryPatterns = [
    { pattern: /(?:i (?:always|usually|prefer|like to|tend to))\s+(.+)/i, key: 'preferences' },
    { pattern: /(?:i'm |i am )(?:a |an )?(vegetarian|vegan|gluten.free|dairy.free|allergic to .+)/i, key: 'dietary_needs' },
    { pattern: /(?:my (?:dog|pet|partner|wife|husband|kid|family))/i, key: 'camping_companions' },
  ]

  for (const { pattern, key } of memoryPatterns) {
    const match = userMessage.match(pattern)
    if (match) {
      await prisma.agentMemory.upsert({
        where: { key },
        update: { value: match[0] },
        create: { key, value: match[0] },
      })
    }
  }
}
```

The `buildContextWindow` function returns an array compatible with the Anthropic Messages API `messages` parameter. The `extractAndSaveMemory` function is intentionally simple — it pattern-matches user messages for memory-worthy facts and upserts them. This can be made more sophisticated later (e.g., ask Claude to extract memories) but starts simple to avoid latency.
