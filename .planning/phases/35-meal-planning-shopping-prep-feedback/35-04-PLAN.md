Compute the meal plan status string in page.tsx (server-side) and pass to DashboardClient:

```typescript
function getMealPlanStatus(trip: typeof upcomingTrip): string | null {
  if (!trip) return null
  if (!trip.mealPlanGeneratedAt) return 'No meal plan yet \u2014 tap to generate'
  const items = trip.mealPlan?.shoppingItems ?? []
  if (items.length === 0) return 'Meal plan ready \u2014 shopping list pending'
  const unchecked = items.filter(i => !i.checked).length
  if (unchecked === 0) return 'All stocked up \u2014 ready to roll'
  return `Shopping list ready \u2014 ${unchecked} items to get`
}
```

Pass to DashboardClient:
```typescript
<DashboardClient
  stats={...}
  recentGear={...}
  upcomingTrip={upcomingTrip ? {
    ...existingProps,
    mealPlanStatus: getMealPlanStatus(upcomingTrip),
  } : null}
/>
```
